# 拓扑排序

使用链式前向星存图

```c++
#include <iostream>
#include <cstring>
#include <queue>
using namespace std;

struct edge {
    int to, next;//终点与下一个
};

edge edg[50];//边
int n, m, head[50], in_degree[50];//入度

int main() {
    memset(head, -1, sizeof(head));//初始化head数组
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int a, b;
        cin >> a >> b;
        edg[i].to = b;//记录终点
        edg[i].next = head[a];//记录起点
        head[a] = i;
        in_degree[b]++;//入度
    }
    queue<int> que;
    for (int i = 1; i <= n; i++) {//先找到所有的入度为0的点
        if (in_degree[i] == 0) {
            que.push(i);
        }
    }
    while (!que.empty()) {
        int temp = que.front();//弹出一个点
        que.pop();
        cout << temp << " ";
        for (int i = head[temp]; i != -1; i = edg[i].next) {
            int e = edg[i].to;//边的终点
            in_degree[e]--;//入度--
            if (in_degree[e] == 0) {//如果变为0了，进队列
                    que.push(e);
            }
        }
    }
    return 0;
}
```

## [HZOJ-641拓扑排序](http://oj.haizeix.com/problem/641)

多个拓扑序列，尽量让小数在大数前面

```c++
#include <iostream>
#include <cstring>
#include <queue>
using namespace std;

struct edge {
    int to, next;
};

struct my_int {//重载小于号，按照自己的数据类型走
    int num;
    bool operator< (const my_int & b) const {
        return this->num > b.num;
    }
};

edge edg[400005];
int n, m, head[2005], in_degree[2005];

int main() {
    memset(head, -1, sizeof(head));
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int a, b;
        cin >> a >> b;
        edg[i].to = b;
        edg[i].next = head[a];
        head[a] = i;
        in_degree[b]++;
    }
    priority_queue<my_int> que;//优先队列重载小于号
    for (int i = 1; i <= n; i++) {
        if (in_degree[i] == 0) {
            que.push((my_int){i});
        }
    }
    int f = 0;
    while (!que.empty()) {
        int temp = que.top().num;
        que.pop();
        if (f == 1) {
            cout << " ";
        }
        f = 1;
        cout << temp;
        for (int i = head[temp]; i != -1; i = edg[i].next) {
            int e = edg[i].to;
            in_degree[e]--;
            if (in_degree[e] == 0) {
                que.push((my_int){e});
            }
        }
    }
    return 0;
}
```

## [HZOJ-640食物链计数](http://oj.haizeix.com/problem/640)

![image-20200729162932735](http://test-fangsong-imgsubmit.oss-cn-beijing.aliyuncs.com/img/image-20200729162932735.png)

```c++
#include <iostream>
#include <cstring>
#include <queue>
using namespace std;

struct edge {
    int to, next;
};

edge edg[500005];
int n, m, head[5005], ans[5005], in_degree[5005], out_degree[5005];

int main() {
    memset(head, -1, sizeof(head));
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int a, b;
        cin >> a >> b;
        edg[i].to = b;
        edg[i].next = head[a];
        head[a] = i;
        in_degree[b]++;//出度入度统计
        out_degree[a]++;
    }
    queue<int> que;
    for (int i = 1; i <= n; i++) {//起点先加入队列
        if (in_degree[i] == 0) {
            que.push(i);
            ans[i] = 1;
        }
    }
    while (!que.empty()) {
        int t = que.front();
        que.pop();
        for (int i = head[t]; i != -1; i = edg[i].next) {
            int e = edg[i].to;
            in_degree[e]--;//终点的入度--
            ans[e] += ans[t];//终点的个数加上起点的个数
            ans[e] %= 100000007;
            if (in_degree[e] == 0) {//终点入度为0就入队
                que.push(e);
            }
        }
    }
    int fin = 0;
    for (int i = 1; i <= n; i++) {
        if (out_degree[i] == 0) {
            fin += ans[i];//统计最后总个数
            fin %= 100000007;
        }
    }
    cout << fin << endl;
    return 0;
}
```

## [HZOJ-635神经网络](http://oj.haizeix.com/problem/635)

![image-20200729163736632](http://test-fangsong-imgsubmit.oss-cn-beijing.aliyuncs.com/img/image-20200729163736632.png)

```c++
#include <iostream>
#include <cstring>
#include <queue>
using namespace std;

struct edge {
    int to, val, next;
};


edge edg[10005];
int n, m, head[105], in_degree[105], out_degree[105], c[105], u[105];
//cu是属性
int main() {
    memset(head, -1, sizeof(head));
    queue<int> que;
    cin >> n >> m;
    for (int i = 1; i <= n; i++) {
        cin >> c[i] >> u[i];
        if (c[i] != 0) {//不为0的为起点
            que.push(i);
        }
    }
    for (int i = 0; i < m; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        edg[i].to = b;
        edg[i].val = c;
        edg[i].next = head[a];
        head[a] = i;
        in_degree[b]++;
        out_degree[a]++;
    }
    while (!que.empty()) {
        int t = que.front();
        que.pop();
        for (int i = head[t]; i != -1; i = edg[i].next) {
            int e = edg[i].to, v = edg[i].val;//终点与权值
            if (c[t] > 0) {
                c[e] += v * c[t];
            }
            in_degree[e]--;
            if (in_degree[e] == 0) {
                que.push(e);
                c[e] -= u[e];//减掉本身的阈值
            }
        }
    }
    int f = 0;
    for (int i = 1; i <= n; i++) {
        if (out_degree[i] == 0 && c[i] > 0) {
            cout << i << " " << c[i] << endl;
            f = 1;
        }
    }
    if (f == 0) {
        cout << "NULL" << endl;
    }
    return 0;
}
```

## [HZOJ-636旅行计划](http://oj.haizeix.com/problem/636)

```c++
#include <iostream>
#include <cstring>
#include <queue>
using namespace std;

struct edge {
    int to, next;
};

edge edg[200005];
int n, m, head[100005], in_degree[100005], ans[100005];

int main() {
    memset(head, -1, sizeof(head));
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        int a, b;
        cin >> a >> b;
        edg[i].to = b;
        edg[i].next = head[a];
        head[a] = i;
        in_degree[b]++;
    }
    queue<int> que;
    for (int i = 1; i <= n; i++) {
        if (in_degree[i] == 0) {
            ans[i] = 1;
            que.push(i);
        }
    }
    while (!que.empty()) {
        int t = que.front();
        que.pop();
        for (int i = head[t]; i != -1; i = edg[i].next) {//遍历以它为起点的所有边
            int e = edg[i].to;
            ans[e] = max(ans[e], ans[t] + 1);//本身， 起点浏览的城市+到这1
            in_degree[e]--;
            if (in_degree[e] == 0) {
                que.push(e);
            }
        }
    }
    for (int i = 1; i <= n; i++) {
        cout << ans[i] << endl;
    }
    return 0;
}
```

## [HZOJ-638最长路](http://oj.haizeix.com/problem/638)

```c++
#include <iostream>
#include <cstring>
#include <queue>
using namespace std;

struct edge {
    int to, val, next;
};

edge edg[50005];
int n, m, head[1505], ans[1505], in_degree[1505], out_degree[1505];

int check() {//成环不连通
    int mark[1505] = {0, 1};
    queue<int> que;
    que.push(1);
    while (!que.empty()) {
        int t = que.front();
        que.pop();
        for (int i = head[t]; i != -1; i = edg[i].next) {
            int e = edg[i].to;
            if (e == n) {
                return 1;
            }
            if (mark[e] == 0) {
                mark[e] = 1;
                que.push(e);
            }
        }
    }
    return 0;
}

int main() {
    memset(head, -1, sizeof(head));
    cin >> n >> m;
    for (int i = 1; i <= m; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        edg[i].to = b;
        edg[i].val = c;
        edg[i].next = head[a];
        head[a] = i;
        in_degree[b]++;
        out_degree[a]++;
    }
    if (check() == 0) {
        cout << -1 << endl;
        return 0;
    }
    queue<int> que;
    for (int i = 1; i <= n; i++) {
        ans[i] = -2100000000;
        if (in_degree[i] == 0) {
            que.push(i);
        }
    }
    ans[1] = 0;
    while (!que.empty()) {
        int temp = que.front();
        que.pop();
        for (int i = head[temp]; i != -1; i = edg[i].next) {
            int e = edg[i].to;
            ans[e] = max(ans[e], ans[temp] + edg[i].val);
            in_degree[e]--;
            if (in_degree[e] == 0) {
                que.push(e);
            }
        }
    }
    cout << ans[n] << endl;
    return 0;
}
```

## [HZOJ-637排序](http://oj.haizeix.com/problem/637)

两个及以上入度为0的点称谓普通点;

遍历完图还有入度则有环;

除了普通和有环就是有解。

![image-20200730222605474](http://test-fangsong-imgsubmit.oss-cn-beijing.aliyuncs.com/img/image-20200730222605474.png)

```cpp
#include <iostream>
#include <queue>
#include <cstring>
using namespace std;

struct edge {//链式前向星
    int to, next;
};

edge edg[10000];//边集
int n, m, edg_cnt, ans_cnt, head[30], in_degree[30];
//edg_cnt存到了第几条边
//ans_cnt答案存到了多少个
char ans[30];//存字符


void insert_edg(int a, int b, int x) {
    edg[x].to = b;
    edg[x].next = head[a];
    head[a] = x;
    edg_cnt = x + 1;
    in_degree[b]++;//入度
}

int check() {
    int temp_in[30];//临时的入度
    ans_cnt = 0;
    queue<int> que;
    for (int i = 1; i <= n; i++) {//存到临时的入度
        temp_in[i] = in_degree[i];
        if (temp_in[i] == 0) {
            que.push(i);
        }
    }
    int f1 = que.size(), f2 = 0; //起点的点的数量 如果不为1 说明答案不确定
    while (!que.empty()) {
        int t = que.front();
        int f3 = 0;
        que.pop();
        ans[++ans_cnt] = t + 'A' - 1;
        for (int i = head[t]; i != -1; i = edg[i].next) {
            int e = edg[i].to;
            temp_in[e]--;
            if (temp_in[e] == 0) {
                if (f3 == 1) {
                    f3 = 2; //f3 为 2 时 说明答案不确定
                } else if (f3 == 0) {
                    f3 = 1;
                }
                que.push(e);
            }
        }
        if (f3 == 2) {
            f2 = 2;
        }
    }
    for (int i = 1; i <= n; i++) {
        if (temp_in[i] != 0) {//成环了
            return 1;
        }
    }
    if (f1 != 1 || f2 == 2) {//普通
        return 0;
    }
    return 2;
}

int main() {
    memset(head, -1, sizeof(head));
    cin >> n >> m;
    for (int i = 0; i < m; i++) {
        char a, b;
        cin >> a >> b >> b;// A < B
        insert_edg(a - 'A' + 1, b - 'A' + 1, i);//插入A -> B, 编号 i 
        int t = check(); // 0 normal   1 wrong   2 ans
        if (t == 1) {
            cout << "Inconsistency found after " << i + 1 << " relations." << endl;
            return 0;
        } else if (t == 2) {
            cout << "Sorted sequence determined after " << i + 1 << " relations: ";
            for (int i = 1; i <= ans_cnt; i++) {
                cout << ans[i];
            }
            cout << "." << endl;
            return 0;
        }
    }
    cout << "Sorted sequence cannot be determined." << endl;
    return 0;
}
```
